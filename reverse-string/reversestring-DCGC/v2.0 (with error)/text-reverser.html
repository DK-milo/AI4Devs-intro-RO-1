// Wait for the DOM to be fully loaded document.addEventListener('DOMContentLoaded', function() { // Get references to DOM elements const inputText = document.getElementById('inputText'); const reverseButton = document.getElementById('reverseButton'); const clearButton = document.getElementById('clearButton'); const reversedText = document.getElementById('reversedText'); // Add event listener for input changes inputText.addEventListener('input', function() { const text = inputText.value; // Only reverse if text length is greater than 3 if (text.length > 3) { const reversed = text.split('').reverse().join(''); reversedText.textContent = reversed; } else { reversedText.textContent = text.length > 0 ? 'Type at least 4 characters...' : ''; } }); // Add event listener for the clear button clearButton.addEventListener('click', clearAll); // Function to reverse the input text function reverseInputText() { const text = inputText.value; if (text.trim() === '') { reversedText.textContent = 'Please enter some text to reverse.'; return; } // Reverse the text using Array methods const reversed = text.split('').reverse().join(''); // Display the reversed text reversedText.textContent = reversed; } // Function to clear input and output function clearAll() { inputText.value = ''; reversedText.textContent = ''; inputText.focus(); } });
